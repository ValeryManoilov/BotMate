Работа с Config.py

# Импорт функции randint
from random import randint

# Создание класса Config
class Config:
    # Объявление токена
    BOT_TOKEN = "Ваш токен из BotFather"
    # Создание конструктора класса
    def __init__(self, TRIES = 1, RANDOM_NUM = randint(1, 100)):
        # Кол-во попыток
        self.TRIES = TRIES
        # Случайное число
        self.RANDOM_NUM = RANDOM_NUM

    # Функция, меняющая количество попыток
    def set_tries(self, TRIES):
        self.TRIES = TRIES

    # Функция, меняющая случайное число
    def reset_random_num(self):
        self.RANDOM_NUM = randint(1, 100)

Работа с Bot_api.py

# Импорт составляющих библиотеки aiogram
from aiogram import Bot, Dispatcher, types
from aiogram.utils import executor
# Импорт класса Config
from Config import Config

# Объявления объекта класса Config
config = Config()

# Объявление переменной токена
botToken = config.BOT_TOKEN

# Объявления объекта класса Config
bot = Bot(token=botToken)
# Подключение диспетчера
dp = Dispatcher(bot)
# Функция, которая обрабатывает команду /start
@dp.message_handler(commands=["start"])
async def StartGame(message: types.Message):
    await message.answer("Добро пожаловать в игру 'угадай число'.
Введи /number <число от 1 до 100>")

# Функция, которая обрабатывает команду /number
@dp.message_handler(commands=["number"])
async def GuessNumber(message: types.Message):

    # Объявление случайного числа
    randonNum = config.RANDOM_NUM
    # Объявление переменной попыток
    tries = config.TRIES

    # Проверка условия соответствия числа пользователя
# с числом игры
    # Если число пользователя - число игры
    if (int(message.text[8:]) == randonNum):
        await message.answer(f"Поздравляю! Количество
попыток: {tries}")
        await message.answer(f"Началась новая игра")
        # Сброс попыток
        config.set_tries(0)
        # Переопределение случайного числа
        config.reset_random_num()

    # Иначе
    else:
        if (int(message.text[8:]) < randonNum):
            await message.answer(f"Ваше число меньше загаданного")
        elif (int(message.text[8:]) > randonNum):
            await message.answer(f"Ваше число больше загаданного")
        config.set_tries((tries + 1))
# Функция, которая позволяет запускать бота в Telegram
if __name__ == '__main__':
    executor.start_polling(dispatcher=dp, skip_updates=True)
